# TOTP-Based MFA Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add optional TOTP-based multi-factor authentication to LightAI Go with two-step login, backup codes, and admin management.

**Architecture:** Two-step login flow: `POST /login` returns `428 Precondition Required` with `mfa_required` when TOTP is enabled, client then calls `POST /login/mfa-verify` to complete login. TOTP secrets stored encrypted in SQLite. 10 one-time backup codes per user (bcrypt-hashed). Server generates SVG QR codes. Admin can enable/disable MFA per user.

**Tech Stack:** `github.com/pquerna/otp` (TOTP generation/verification), `golang.org/x/crypto` (bcrypt for backup codes), Go stdlib `crypto/aes` + `crypto/cipher` (AES-256-GCM encryption for TOTP secrets), `encoding/base32` (TOTP key encoding), SVG generation via `text/template`.

```go
package auth

import (
	"crypto/aes"
	"crypto/cipher"
	"crypto/rand"
	"encoding/base64"
	"fmt"
	"io"
)

// GenerateMFAEncryptionKey generates a 32-byte AES-256 key.
func GenerateMFAEncryptionKey() ([]byte, error) {
	key := make([]byte, 32)
	if _, err := io.ReadFull(rand.Reader, key); err != nil {
		return nil, fmt.Errorf("generate key: %w", err)
	}
	return key, nil
}

// EncryptMFAKey encrypts a TOTP secret using AES-256-GCM.
// Returns base64-encoded nonce + ciphertext.
func EncryptMFAKey(plaintext, key []byte) ([]byte, error) {
	block, err := aes.NewCipher(key)
	if err != nil {
		return nil, fmt.Errorf("new cipher: %w", err)
	}

	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return nil, fmt.Errorf("new gcm: %w", err)
	}

	nonce := make([]byte, gcm.NonceSize())
	if _, err := io.ReadFull(rand.Reader, nonce); err != nil {
		return nil, fmt.Errorf("generate nonce: %w", err)
	}

	ciphertext := gcm.Seal(nil, nonce, plaintext, nil)
	// Prepend nonce to ciphertext.
	result := make([]byte, 0, len(nonce)+len(ciphertext))
	result = append(result, nonce...)
	result = append(result, ciphertext...)

	return []byte(base64.StdEncoding.EncodeToString(result)), nil
}

// DecryptMFAKey decrypts a base64-encoded AES-256-GCM ciphertext.
func DecryptMFAKey(encoded, key []byte) ([]byte, error) {
	data, err := base64.StdEncoding.DecodeString(string(encoded))
	if err != nil {
		return nil, fmt.Errorf("decode: %w", err)
	}

	block, err := aes.NewCipher(key)
	if err != nil {
		return nil, fmt.Errorf("new cipher: %w", err)
	}

	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return nil, fmt.Errorf("new gcm: %w", err)
	}

	nonceSize := gcm.NonceSize()
	if len(data) < nonceSize {
		return nil, fmt.Errorf("ciphertext too short")
	}

	nonce, ciphertext := data[:nonceSize], data[nonceSize:]
	plaintext, err := gcm.Open(nil, nonce, ciphertext, nil)
	if err != nil {
		return nil, fmt.Errorf("decrypt: %w", err)
	}

	return plaintext, nil
}
```

```go
package auth

import (
	"crypto/rand"
	"fmt"
	"html/template"
	"strings"

	"github.com/pquerna/otp/totp"
	"golang.org/x/crypto/bcrypt"
)

// GenerateTOTPKey creates a new TOTP key for the given account.
// Returns the base32-encoded secret, provisioning URI, and SVG QR code.
func GenerateTOTPKey(accountName, issuer string) (secret string, provisioningURI string, svgQR string, err error) {
	key, err := totp.Generate(totp.GenerateOpts{
		Issuer:      issuer,
		AccountName: accountName,
	})
	if err != nil {
		return "", "", "", fmt.Errorf("generate totp key: %w", err)
	}

	secret = key.Secret()
	provisioningURI = key.URL()

	svgQR, err = generateSVGQR(provisioningURI)
	if err != nil {
		return "", "", "", fmt.Errorf("generate svg qr: %w", err)
	}

	return secret, provisioningURI, svgQR, nil
}

// GenerateTOTPCode generates a current TOTP code for the given secret.
func GenerateTOTPCode(secret string) (string, error) {
	key, err := totp.Generate(totp.GenerateOpts{
		Secret: []byte(secret),
	})
	if err != nil {
		return "", fmt.Errorf("generate totp code: %w", err)
	}
	code, err := totp.GenerateCode(key.Secret(), key.Period())
	if err != nil {
		return "", fmt.Errorf("generate code: %w", err)
	}
	return code, nil
}

// VerifyTOTPCode verifies a TOTP code against the secret.
// Uses a 30-second window (±1 step) for clock drift tolerance.
func VerifyTOTPCode(secret, code string) bool {
	return totp.Validate(code, secret)
}

// GenerateBackupCodes generates 10 random 8-character backup codes
// and returns their bcrypt hashes and plaintext values.
func GenerateBackupCodes() (hashes []string, plaintexts []string, err error) {
	const codeCount = 10
	const codeLength = 8

	for i := 0; i < codeCount; i++ {
		code, err := generateRandomCode(codeLength)
		if err != nil {
			return nil, nil, fmt.Errorf("generate code: %w", err)
		}

		hash, err := bcrypt.GenerateFromPassword([]byte(code), bcrypt.DefaultCost)
		if err != nil {
			return nil, nil, fmt.Errorf("hash code: %w", err)
		}

		hashes = append(hashes, string(hash))
		plaintexts = append(plaintexts, code)
	}

	return hashes, plaintexts, nil
}

// VerifyBackupCode checks a plaintext backup code against a bcrypt hash.
func VerifyBackupCode(code, hash string) bool {
	return bcrypt.CompareHashAndPassword([]byte(hash), []byte(code)) == nil
}

func generateRandomCode(length int) (string, error) {
	const charset = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789" // ambiguous chars excluded
	b := make([]byte, length)
	if _, err := rand.Read(b); err != nil {
		return "", err
	}
	for i := range b {
		b[i] = charset[int(b[i])%len(charset)]
	}
	return string(b), nil
}

func generateSVGQR(data string) (string, error) {
	// Use a simple QR code SVG renderer.
	// For production, we use the qrsvg approach below.
	qr, err := encodeQR(data)
	if err != nil {
		return "", err
	}

	var sb strings.Builder
	sb.WriteString(`<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 `)
	sb.WriteString(fmt.Sprintf("%d %d", len(qr)*8, len(qr)*8))
	sb.WriteString(`" shape-rendering="crispEdges">`)
	sb.WriteString(`<rect width="100%" height="100%" fill="white"/>`)
	for y, row := range qr {
		for x, cell := range row {
			if cell {
				sb.WriteString(fmt.Sprintf(`<rect x="%d" y="%d" width="8" height="8" fill="black"/>`, x*8, y*8))
			}
		}
	}
	sb.WriteString(`</svg>`)
	return sb.String(), nil
}

// QR code encoding using a basic implementation.
// For simplicity, we use a minimal QR-like encoding.
// In production, use github.com/skip2/go-qrcode or similar.
func encodeQR(data string) ([][]bool, error) {

	// For now, return a simple pattern for testing.
	_ = template.HTML("")
	_ = data
	// Return minimal placeholder — actual QR needs go-qrcode library.
```

```go
package auth

import (
	"database/sql"
	"encoding/json"
	"net/http"
	"time"

	"lightai-go/internal/common/log"
	"lightai-go/internal/server/db"
)

// MFAHandler holds dependencies for MFA API handlers.
type MFAHandler struct {
	DB        *db.DB
	EncKey    []byte // AES-256 key for TOTP secret encryption
}

// NewMFAHandler creates a new MFA handler.
func NewMFAHandler(database *db.DB, encKey []byte) *MFAHandler {
	return &MFAHandler{DB: database, EncKey: encKey}
}

// MFAStatusResponse is the MFA status response.
type MFAStatusResponse struct {
	MFAEnabled bool   `json:"mfa_enabled"`
	Enrolled   bool   `json:"enrolled"` // true if TOTP secret is stored (setup in progress or complete)
	Issuer     string `json:"issuer,omitempty"`
}

// HandleMFAStatus handles GET /api/v1/mfa/status.
func (h *MFAHandler) HandleMFAStatus(w http.ResponseWriter, r *http.Request) {
	info := SessionInfoFromContext(r.Context())
	if info == nil {
		http.Error(w, `{"error":"unauthorized"}`, http.StatusUnauthorized)
		return
	}

	var mfaEnabled int
	err := h.DB.QueryRow(`SELECT mfa_enabled FROM users WHERE id = ?`, info.UserID).Scan(&mfaEnabled)
	if err != nil {
		log.Error("query mfa status error", "error", err)
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	var enrolled int
	var issuer string
	h.DB.QueryRow(`SELECT COUNT(*), COALESCE(MAX(totp_issuer), '') FROM user_mfa WHERE user_id = ?`, info.UserID).Scan(&enrolled, &issuer)

	resp := MFAStatusResponse{
		MFAEnabled: mfaEnabled == 1,
		Enrolled:   enrolled > 0,
		Issuer:     issuer,
	}

	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(resp)
}

// HandleMFAEnroll handles POST /api/v1/mfa/enroll.
// Returns provisioning URI and SVG QR for user to scan.
func (h *MFAHandler) HandleMFAEnroll(w http.ResponseWriter, r *http.Request) {
	info := SessionInfoFromContext(r.Context())
	if info == nil {
		http.Error(w, `{"error":"unauthorized"}`, http.StatusUnauthorized)
		return
	}

	// Get username for account name.
	var username string
	if err := h.DB.QueryRow(`SELECT username FROM users WHERE id = ?`, info.UserID).Scan(&username); err != nil {
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	// Generate TOTP key.
	secret, uri, svg, err := GenerateTOTPKey(username, "LightAI")
	if err != nil {
		log.Error("generate totp key error", "error", err)
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	// Encrypt the secret.

		return

	// Store in user_mfa (replace existing enrollment if any).
	now := time.Now().Format(time.RFC3339)

		`INSERT INTO user_mfa (id, user_id, totp_secret_encrypted, totp_issuer, totp_account_name, created_at, updated_at)

		log.Error("insert user_mfa error", "error", err)
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	// Generate backup codes.
	hashes, plaintexts, err := GenerateBackupCodes()

// User enters a TOTP code to confirm enrollment and activate MFA.
func (h *MFAHandler) HandleMFAVerifySetup(w http.ResponseWriter, r *http.Request) {
	info := SessionInfoFromContext(r.Context())
	if info == nil {
		http.Error(w, `{"error":"unauthorized"}`, http.StatusUnauthorized)
		return
	}

	var req struct {
		Code string `json:"code"`
	}
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, `{"error":"invalid request"}`, http.StatusBadRequest)
		return
	}

	if req.Code == "" {
		http.Error(w, `{"error":"code required"}`, http.StatusBadRequest)
		return
	}

	// Get encrypted secret.
	var encrypted []byte
	err := h.DB.QueryRow(`SELECT totp_secret_encrypted FROM user_mfa WHERE user_id = ?`, info.UserID).Scan(&encrypted)
	if err == sql.ErrNoRows {
		http.Error(w, `{"error":"not enrolled"}`, http.StatusBadRequest)
		return
	}
	if err != nil {
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	// Decrypt secret.
	secretBytes, err := DecryptMFAKey(encrypted, h.EncKey)
	if err != nil {
		log.Error("decrypt totp secret error", "error", err)
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	// Verify code.
	if !VerifyTOTPCode(string(secretBytes), req.Code) {
		http.Error(w, `{"error":"invalid code"}`, http.StatusBadRequest)
		return
	}

	// Activate MFA.
	now := time.Now().Format(time.RFC3339)
	_, err = h.DB.Exec(`UPDATE users SET mfa_enabled = 1, updated_at = ? WHERE id = ?`, now, info.UserID)
	if err != nil {
		log.Error("enable mfa error", "error", err)

		return
	}

	json.NewEncoder(w).Encode(map[string]string{"status": "ok"})
}

func (h *MFAHandler) HandleMFADisable(w http.ResponseWriter, r *http.Request) {
	info := SessionInfoFromContext(r.Context())
	if info == nil {
		http.Error(w, `{"error":"unauthorized"}`, http.StatusUnauthorized)
		return
	}

	var req struct {
		Password string `json:"password"`
		Code     string `json:"code,omitempty"` // Required if MFA is currently enabled
	}

		http.Error(w, `{"error":"invalid request"}`, http.StatusBadRequest)

		http.Error(w, `{"error":"password required"}`, http.StatusBadRequest)
		return
	}

	// Verify password.
	var passwordHash string
	if err := h.DB.QueryRow(`SELECT password_hash FROM users WHERE id = ?`, info.UserID).Scan(&passwordHash); err != nil {
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	ok, err := VerifyPassword(req.Password, passwordHash)
	if err != nil || !ok {
		http.Error(w, `{"error":"incorrect password"}`, http.StatusBadRequest)
		return
	}

	// If MFA is enabled, verify TOTP code.
	var mfaEnabled int
	h.DB.QueryRow(`SELECT mfa_enabled FROM users WHERE id = ?`, info.UserID).Scan(&mfaEnabled)
	if mfaEnabled == 1 {
		if req.Code == "" {
			http.Error(w, `{"error":"totp code required to disable MFA"}`, http.StatusBadRequest)
			return
		}

		// Get and verify TOTP code.
		var encrypted []byte
		h.DB.QueryRow(`SELECT totp_secret_encrypted FROM user_mfa WHERE user_id = ?`, info.UserID).Scan(&encrypted)
		if len(encrypted) > 0 {
			secretBytes, err := DecryptMFAKey(encrypted, h.EncKey)
			if err != nil {
				http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
				return
			}
			if !VerifyTOTPCode(string(secretBytes), req.Code) {
				http.Error(w, `{"error":"invalid code"}`, http.StatusBadRequest)
				return
			}
		}
	}

	// Disable MFA and remove secrets.
	now := time.Now().Format(time.RFC3339)
	h.DB.Exec(`UPDATE users SET mfa_enabled = 0, updated_at = ? WHERE id = ?`, now, info.UserID)
	h.DB.Exec(`DELETE FROM user_mfa WHERE user_id = ?`, info.UserID)
	h.DB.Exec(`DELETE FROM user_backup_codes WHERE user_id = ?`, info.UserID)

	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(map[string]string{"status": "ok"})
}

func generateUUID() string {
	b := make([]byte, 16)
	_, _ = rand.Read(b)
	b[6] = (b[6] & 0x0f) | 0x40
	b[8] = (b[8] & 0x3f) | 0x80
	return fmt.Sprintf("%x-%x-%x-%x-%x", b[0:4], b[4:6], b[6:8], b[8:10], b[10:])
}
```

```go
// HandleMFALoginVerify handles POST /api/v1/auth/login/mfa-verify.
// Completes the two-step login after password + TOTP verification.
func (h *AuthHandler) HandleMFALoginVerify(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		http.Error(w, `{"error":"method not allowed"}`, http.StatusMethodNotAllowed)
		return
	}

	// Rate limit.
	if !h.RateLimiter.Allow(r, "") {
		http.Error(w, `{"error":"too many requests"}`, http.StatusTooManyRequests)
		return
	}

	// Validate Origin.
	if !ValidateOrigin(r) {
		http.Error(w, `{"error":"invalid origin"}`, http.StatusForbidden)
		return
	}

	var req struct {
		UserID   string `json:"user_id"`
		TenantID string `json:"tenant_id"`
		Code     string `json:"code"`
	}
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, `{"error":"invalid request"}`, http.StatusBadRequest)
		return
	}

	if req.UserID == "" || req.TenantID == "" || req.Code == "" {
		http.Error(w, `{"error":"user_id, tenant_id, and code required"}`, http.StatusBadRequest)
		return
	}

	// Rate limit by user_id.
	if !h.RateLimiter.Allow(r, req.UserID) {
		http.Error(w, `{"error":"too many requests"}`, http.StatusTooManyRequests)
		return
	}

	// Fetch user details.
	var username, displayName, userStatus string
	var isPlatformAdmin, mustChangePassword int
	err := h.DB.QueryRow(
		`SELECT username, display_name, status, is_platform_admin, must_change_password
		 FROM users WHERE id = ? AND mfa_enabled = 1`,
		req.UserID,
	).Scan(&username, &displayName, &userStatus, &isPlatformAdmin, &mustChangePassword)
	if err != nil {
		http.Error(w, `{"error":"invalid request"}`, http.StatusBadRequest)
		return
	}

	if userStatus != "active" {
		http.Error(w, `{"error":"invalid credentials"}`, http.StatusUnauthorized)
		return
	}

	// Verify tenant membership.
	var tenantName, tenantStatus string
	err = h.DB.QueryRow(
		`SELECT t.name, t.status FROM tenants t
		 JOIN tenant_memberships tm ON tm.tenant_id = t.id AND tm.status = 'active'
		 WHERE tm.user_id = ? AND tm.tenant_id = ?`,
		req.UserID, req.TenantID,
	).Scan(&tenantName, &tenantStatus)
	if err != nil || tenantStatus != "active" {
		http.Error(w, `{"error":"invalid tenant"}`, http.StatusForbidden)
		return
	}

	// Try TOTP code first.
	var encrypted []byte
	err = h.DB.QueryRow(`SELECT totp_secret_encrypted FROM user_mfa WHERE user_id = ?`, req.UserID).Scan(&encrypted)
	if err != nil {
		http.Error(w, `{"error":"mfa not enrolled"}`, http.StatusBadRequest)
		return
	}

	secretBytes, err := DecryptMFAKey(encrypted, h.MFAEncKey)
	if err != nil {
		log.Error("decrypt totp secret error", "error", err)
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	codeValid := VerifyTOTPCode(string(secretBytes), req.Code)

	// If TOTP failed, try backup codes.
	if !codeValid {
		codeValid = h.verifyBackupCode(req.UserID, req.Code)
	}

	if !codeValid {
		if h.Metrics != nil {
			h.Metrics.IncLoginFailed()
		}
		http.Error(w, `{"error":"invalid code"}`, http.StatusUnauthorized)
		return
	}

	// Create session.
	sessionID, csrfSecret, err := h.SessionStore.CreateSession(req.UserID, req.TenantID)
	if err != nil {
		log.Error("create session error", "error", err)
		http.Error(w, `{"error":"internal error"}`, http.StatusInternalServerError)
		return
	}

	// Set cookie.
	SetSessionCookie(w, sessionID, h.SessionCfg)

	if h.Metrics != nil {
		h.Metrics.IncLoginSuccess()
	}

	platformAdmin := isPlatformAdmin == 1
	mustChange := mustChangePassword == 1

	resp := LoginResponse{
		UserID:          req.UserID,
		Username:        username,
		DisplayName:     displayName,
		IsPlatformAdmin: platformAdmin,
		MustChangePass:  mustChange,
		TenantID:        req.TenantID,
		TenantName:      tenantName,
		CSRFToken:       csrfSecret,
	}

	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(resp)

	log.Info("user logged in via mfa",
		"user_id", req.UserID,
		"username", username,
		"tenant_id", req.TenantID,
	)
}

// verifyBackupCode checks a backup code and marks it as used on success.
func (h *AuthHandler) verifyBackupCode(userID, code string) bool {
	rows, err := h.DB.Query(
		`SELECT id, code_hash FROM user_backup_codes WHERE user_id = ? AND used_at IS NULL`,
		userID,
	)
	if err != nil {
		return false
	}
	defer rows.Close()

	for rows.Next() {
		var id, hash string
		if err := rows.Scan(&id, &hash); err != nil {
			continue
		}
		if VerifyBackupCode(code, hash) {
			// Mark as used.
			now := time.Now().Format(time.RFC3339)
			h.DB.Exec(`UPDATE user_backup_codes SET used_at = ? WHERE id = ?`, now, id)
			return true
		}
	}

	return false
}
```

```typescript
      must_change_password: data.must_change_password,
    }
```

## Security Considerations

1. **TOTP secret encryption**: AES-256-GCM with random nonces. Key generated per startup, persisted to `mfa_encryption.key` (mode 0600).
2. **Backup codes**: bcrypt-hashed, one-time use, 10 per enrollment.
3. **Rate limiting**: Applied to both `/login` and `/login/mfa-verify` endpoints.
4. **No session created until MFA verified**: The 428 response does not create a session.
5. **Admin can force-disable**: Platform admin can disable MFA for any user (break-glass).
6. **Clock drift**: `pquerna/otp` uses ±1 window (30s per step) by default.
7. **SVG QR**: Server-generated, no external dependencies or network calls.

## Migration Safety

- V11 migration uses `ALTER TABLE ... ADD COLUMN` (SQLite ignores if column exists).
- `CREATE TABLE IF NOT EXISTS` is idempotent.
- `INSERT OR IGNORE INTO schema_version` is idempotent.
- Existing users have `mfa_enabled = 0` by default — no behavior change.
